33 return;

37 packet.readByte(false);
38 packet.readByte(false);
39 final int slot = packet.readShort(false, ByteModification.ADD);
40 final Item item = player.inventory.get(slot);
41
42 if (ItemDefinition.get(itemId) == null)
43 return;
44
45 player.getCombat().reset();
46
47 if (!player.interfaceManager.isClear())
48 player.interfaceManager.close(false);
49
50 if (player.idle)
51 player.idle = false;
52
53 if (item == null)
54 return;
55
56 if (item.getId() != itemId)
57 return;
58
59 if (PluginManager.getDataBus().publish(player, new DropItemEvent(item, slot, player.getPosition().copy())))
60 return;
61
62 if (ItemActionRepository.drop(player, item)) {
63 if (PlayerRight.isDeveloper(player)) {
64 player.send(new SendMessage(String.format("[%s]: item=%d amount=%d slot=%d", ItemActionRepository.class.getSimpleName(), item.getId(), item.getAmount(), slot)));
65 }
66 return;
67 }
68
69 boolean inWilderness = Area.inWilderness(player);
70 if (inWilderness && item.getValue(PriceType.VALUE) >= 500_000) {
71 player.dialogueFactory.sendStatement("This is a valuable item, are you sure you want to", "drop it? In a PvP area, this item will be seen", "by everyone when dropped.");
72 player.dialogueFactory.sendOption("Yes, drop it.", () -> {
73 player.inventory.remove(item, slot, true);
74 GroundItem.createGlobal(player, item);
75 }, "Nevermind", () -> player.dialogueFactory.clear());
76 player.dialogueFactory.execute();

79 player.inventory.remove(item, slot, true);

81 return;
82 }
83
