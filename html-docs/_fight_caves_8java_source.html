81 npcs.add(TzKet);
82 TzKet.getCombat().attack(player);

84 return;

86
87 npcs.remove(dead);
88 remove(dead);
89 tokkuls += Utility.random(100, 500);
90 if (npcs.isEmpty()) {

92 if (wave == null) {
93 completed = true;
94 player.send(new SendMessage("You have finished the activity!"));
