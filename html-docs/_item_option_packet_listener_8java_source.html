60 private static void handleFirstOption(Player player, GamePacket packet) {
61 final int interfaceId = packet.readShort(ByteOrder.LE, ByteModification.ADD);
62 final int slot = packet.readShort(false, ByteModification.ADD);
63 final int id = packet.readShort(ByteOrder.LE);
64
65 switch (interfaceId) {
66 case 57716:
67 player.forClan(channel -> channel.getShowcase().select(player, id, slot));
68 break;
69 case InterfaceConstants.INVENTORY_INTERFACE:
70 final Item item = player.inventory.get(slot);
71
72 if (item == null || item.getId() != id) {
73 return;
74 }

77 return;
