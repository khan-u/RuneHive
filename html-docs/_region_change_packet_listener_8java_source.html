13 * The {@link GamePacket}'s responsible for changing a players region. Used when
14 * a player enters a new map region or when the map region has been successfully
15 * loaded.
16 *
17 * @author Daniel

19@PacketListenerMeta({ClientPackets.LOADED_REGION, ClientPackets.ENTER_REGION})
20public class RegionChangePacketListener implements PacketListener {
21
22 @Override
23 public void handlePacket(Player player, GamePacket packet) {
24 switch (packet.getOpcode()) {
25 case ClientPackets.ENTER_REGION:
26 int a = packet.readInt();
27 if (player.debug && PlayerRight.isDeveloper(player)) {
28 player.send(new SendMessage("[REGION] Entered new region: " + a, MessageColor.DEVELOPER));
29 }
30 if (a != 0x3f008edd) {
31 player.getEvents().setLogOut(true);
32 }
33 break;
34
35 case ClientPackets.LOADED_REGION:
36 player.getEvents().setLoadRegion(true);
37 break;
38 }
39 }
