107 for (Player bot : World.getPlayers()) {
108 if (!bot.isBot) {

110 }
111 for (String nameList : BOT_NAMES) {
112 if (bot.getName().equalsIgnoreCase(nameList)) {
113 names.remove(nameList);
114 }
115 }
116 }
117 return names;
